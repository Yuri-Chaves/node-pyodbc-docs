# MIN

Returns the smallest value within the selected column

```ts
import { odbc } from "./odbc";
import { TUsers } from "./types";

const min = await odbc.aggregate<TUsers, { youngest: number }>({
  fn: "MIN",
  column: "age",
  table: "users",
  alias: "youngest"
});

console.log(min);
/*
{
  youngest: 29
}
```

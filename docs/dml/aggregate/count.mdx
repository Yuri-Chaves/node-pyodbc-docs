# COUNT

Returns the number of rows in a set

```ts
import { odbc } from "./odbc";
import { TUsers } from "./types";

const count = await odbc.aggregate<TUsers, Array<TUsers>>({
  fn: "COUNT",
  column: "*",
  table: "users",
  distinct: true,
});

console.log(count);
```

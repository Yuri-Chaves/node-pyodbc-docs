# SUM

Returns the total sum of a numerical column

```ts
import { odbc } from "./odbc";
import { TUsers } from "./types";

const sum = await odbc.aggregate<TUsers, { total: number }>({
  fn: "SUM",
  column: "age",
  table: "users",
  alias: "total",
  expression: "* 2", // Multiply 2 times each value
});

console.log(sum);
/*
{
  total: 180
}
*/
```

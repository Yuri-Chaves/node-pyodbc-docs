# COUNT

Retorna a quantidade de linhas em um conjunto de resultados.

```ts
import { odbc } from "./odbc";
import { TUsuarios } from "./types";

const count = await odbc.aggregate<TUsuarios, Array<TUsuarios>>({
  fn: "COUNT",
  column: "*",
  table: "usuarios",
  distinct: true,
});

console.log(count);
```
